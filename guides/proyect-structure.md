# Estructura del proyecto

guides/ <- Guias de desarrollador
routes/ <- Rutas de la API
	auth.controller.js <- Ruta para la gestion de usuarios (register, login, actualizar)
	post.controller.js <- Ruta para la gestion de posts (creacion, eliminacion)
	user.route.js <- Ruta para usuarios (obtenerlos)
.gitignore <- Archivo de configuracion de git
app.js <- Archivo principal
CHANGELOG.md <- Registro de actualizaciones del backend
client.js <- Conexion con supabase
package-lock.json <- Paquetes de node
package.json <- Funcionalidad de node
README.md <- Informacion del proyecto
